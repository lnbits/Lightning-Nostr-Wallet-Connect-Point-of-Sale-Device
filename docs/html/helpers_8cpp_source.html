120 mpz_init(exponent);
121 mpz_init(temp);
122
123 // Legendre symbol: (n/p) = n^((p-1)/2) mod p
124 mpz_sub_ui(temp, p, 1); // temp = p - 1
125 mpz_fdiv_q_ui(exponent, temp, 2); // exponent = (p - 1) / 2
