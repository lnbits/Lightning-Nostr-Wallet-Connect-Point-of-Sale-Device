49 static void wifiTask(void *parameter) {
50 Serial.println("WiFi task started");
51 while (true) {
52 wifi_command_t command;
53 if (xQueueReceive(wifi_command_queue, &command, portMAX_DELAY)) {
54 Serial.print("WiFi task received command: ");
55 Serial.println(command.type);
56
57 switch (command.type) {
58 case WIFI_SCAN: {
59 Serial.println("Starting WiFi scan...");
60
61 // Stop auto-reconnect and put WiFi in scan mode
62 WiFi.setAutoReconnect(false);
63 WiFi.disconnect(true); // Disconnect and clear config
64 delay(1000); // Wait for clean disconnect
65
66 // Set WiFi to station mode for scanning
67 WiFi.mode(WIFI_STA);
68 delay(100);
69
70 int n = WiFi.scanNetworks();
71 Serial.print("Scan completed, found ");
72 Serial.print(n);
73 Serial.println(" networks");
74
75 wifi_scan_result_t result;
76
77 // Check for scan errors (negative return values)
78 if (n < 0) {
79 Serial.print("WiFi scan failed with error code: ");
80 Serial.println(n);
81 Serial.println("Retrying scan in 1 second...");
82 delay(1000);
83
84 // Retry scan once
85 n = WiFi.scanNetworks();
86 Serial.print("Retry scan found ");
87 Serial.print(n);
88 Serial.println(" networks");
89 }
90
91 if (n < 0) {
92 Serial.println("Scan failed after retry, returning empty results");
93 result.network_count = 0;
94 } else {
95 result.network_count = (n > 9) ? 9 : n;
96
97 for (int i = 0; i < result.network_count; i++) {
98 strncpy(result.ssids[i], WiFi.SSID(i).c_str(), 32);
99 result.ssids[i][32] = '\0';
100 result.rssi[i] = WiFi.RSSI(i);
101 result.encrypted[i] = (WiFi.encryptionType(i) != WIFI_AUTH_OPEN);
102 }
103 }
104
105 // Re-enable auto-reconnect for normal operation
106 WiFi.setAutoReconnect(true);
107
108 if (wifi_scan_result_queue != NULL) {
109 if (xQueueSend(wifi_scan_result_queue, &result, 0) == pdTRUE) {
110 Serial.println("Scan results sent to queue successfully");
111 } else {
112 Serial.println("Failed to send scan results to queue");
113 }
114 }
115 break;
116 }
117 case WIFI_CONNECT:
118 Serial.println("Connecting to WiFi...");
119 WiFi.begin(command.ssid, command.password);
120 break;
121 case WIFI_DISCONNECT:
122 Serial.println("Disconnecting from WiFi...");
123 WiFi.disconnect(true);
124 break;
125 case WIFI_STOP_SCAN:
126 Serial.println("Stopping WiFi scan...");
127 break;
128 }
129 }
130 vTaskDelay(pdMS_TO_TICKS(100));
131 }
